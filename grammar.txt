fn (int, int) swap (int x, int y) {
    int temp = x;
    x = y;
    y = temp;
    ret (x, y);
}
     
fn main {
    int a = 5;
    int b = 6;
    (a, b) = swap(a, b);
}
    
START = DEF_LIST
DEF_LIST    -> FUNC_DEF DEF_LIST
             | ε 
FUNC_DEF    -> fn RET_LIST id PARAM_LIST STAT_LIST
RET_LIST    -> ( TYPE_NAME RET_LIST' 
             | ε 
RET_LIST'   -> comma TYPE_NAME RET_LIST' 
             | )
PARAM_LIST  -> ( VAR_DEF PARAM_LIST' 
             | ε
PARAM_LIST' -> comma VAR_DEF PARAM_LIST'
             | ) 
             | ε 
VAR_DEC     -> TYPE_NAME id
TYPE_NAME   -> int
             | float
STAT_LIST   -> { STATEMENT STAT_LIST'
             | ε
STAT_LIST'  -> STATEMENT STAT_LIST'
             | }
STATEMENT   -> DEC_STAT | ASSIGN_STAT
DEC_STAT    -> VAR_DEC DEC_STAT'
DEC_STAT'   -> ;
             | = EXPRESSION 
EXPRESSION  -> AND_TERM EXPRESSION'
EXPRESSION' -> | EXPRESSION'
             | ε
AND_TERM    -> EQ_TERM AND_TERM'
AND_TERM'   -> & AND_TERM'
             | ε
EQ_TERM     -> ADDSUB_TERM EQ_TERM'
EQ_TERM'    -> == EQ_TERM'
             | ε
ADDSUB_TERM -> MULDIV_TERM ADDSUB_TERM'
ADDSUB_TERM'-> + ADDSUB_TERM'
             | - ADDSUB_TERM'
             | ε
MULDIV_TERM -> VALUE_TERM MULDIV_TERM'
MULDIV_TERM'-> * MULDIV_TERM'
             | / MULDIV_TERM'
             | ε 
VALUE_TERM  -> id | INT_LIT | FLOAT_LIT | CALL
